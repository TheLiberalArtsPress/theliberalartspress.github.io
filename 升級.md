1. 前端 fetch 與 HMAC 生成範例 (React / JavaScript)
在您的 admin.html 與 index.html 中，請用以下程式碼替換原本的 syncWithGAS 函式。我們使用瀏覽器內建的高效能 Web Crypto API 來生成 Hex 格式的 SHA-256 簽章（不使用 Base64）。

JavaScript
// 🔥 這是您的專屬加密金鑰 (請確保前端與後端的字串完全一致)
const SECRET_KEY = "LAPEN_SECURE_KEY_2026_!@#$"; 

// 1. 生成 HMAC-SHA256 (Hex字串輸出)
async function generateSignature(dataStr, timestamp, secret) {
    const encoder = new TextEncoder();
    const keyData = encoder.encode(secret);
    const msgData = encoder.encode(dataStr + timestamp);

    const cryptoKey = await window.crypto.subtle.importKey(
        "raw", keyData, { name: "HMAC", hash: "SHA-256" }, false, ["sign"]
    );

    const signatureBuffer = await window.crypto.subtle.sign("HMAC", cryptoKey, msgData);
    
    // 將 ArrayBuffer 轉換為 Hex 字串 (例如: a1b2c3d4...)
    const hashArray = Array.from(new Uint8Array(signatureBuffer));
    const hashHex = hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
    return hashHex;
}

// 2. 前端 fetch 核心邏輯
const syncWithGAS = async (action, payloadData = null) => {
    try {
        // 將實際要執行的動作打包成 Payload 字串
        const payloadObj = { action, payload: payloadData };
        const payloadStr = JSON.stringify(payloadObj);
        
        // 取得當前 Unix 毫秒時間戳
        const timestamp = Date.now();
        
        // 產生防竄改簽章: HMAC(secret, payload字串 + timestamp)
        const signature = await generateSignature(payloadStr, timestamp, SECRET_KEY);

        // 僅使用 POST 發送包裹好的安全格式
        const response = await fetch(GAS_URL, {
            method: "POST",
            headers: { "Content-Type": "text/plain;charset=utf-8" },
            body: JSON.stringify({
                payload: payloadStr,
                timestamp: timestamp,
                signature: signature
            })
        });
        
        return await response.json();
    } catch (e) {
        console.error("GAS Sync Error:", e);
        return { status: 'error', msg: e.toString() };
    }
};

在您的 Google Apps Script (Code.gs) 頂部替換以下邏輯。這將完全關閉 doGet，並且嚴格驗證 Timestamp 時效與 Signature。

const SECRET_KEY = "LAPEN_SECURE_KEY_2026_!@#$";

    .setMimeType(ContentService.MimeType.TEXT);

// 2. 嚴格的 POST 安全驗證入口
// ==========================================
function doPost(e) {
  try {
    const requestData = JSON.parse(e.postData.contents);
    const { payload, timestamp, signature } = requestData;

    // [安全防護 1] 缺少必要參數直接拒絕
    if (!payload || !timestamp || !signature) {
      return responseError("Access Denied: Missing security parameters.");
    }

    // [安全防護 2] 驗證 Timestamp 時效 (限制 5 分鐘 = 300,000 毫秒)
    const currentTime = Date.now();
    const requestTime = parseInt(timestamp, 10);
    const timeDiff = Math.abs(currentTime - requestTime);
    
    if (timeDiff > 5 * 60 * 1000) {
      return responseError("Access Denied: Request expired.");
    }

    // [安全防護 3] HMAC-SHA256 驗證 (防竄改)
    const message = payload + String(timestamp);
    
    // GAS 內建的 HMAC SHA256 會輸出 Byte 陣列
    const byteSignature = Utilities.computeHmacSha256Signature(message, SECRET_KEY);
    
    // 將 Byte 陣列轉換為與前端相同的 Hex 小寫字串
    const expectedSignature = byteSignature.map(function(byte) {
      const v = (byte < 0) ? 256 + byte : byte;
      return ("0" + v.toString(16)).slice(-2);
    }).join('');

    // 比對簽章，不符合直接拒絕
    if (expectedSignature !== signature) {
      return responseError("Access Denied: Invalid signature.");
    }

    // ==========================================
    // 3. 驗證成功，放行進入真正的資料處理邏輯
    // ==========================================
    const payloadObj = JSON.parse(payload);
    
    // 呼叫您原本的 processAction 邏輯 (只負責讀寫 Sheet，不須再驗證)
    const result = processAction(payloadObj); 
    
    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return responseError("Server Error: " + err.toString());
  }
}

// 統一錯誤回傳格式
function responseError(msg) {
  return ContentService.createTextOutput(JSON.stringify({ status: "error", msg: msg }))
    .setMimeType(ContentService.MimeType.JSON);
}

🛡️ 升級後達成的安全效果：
防止重放攻擊 (Replay Attack)：駭客就算從 Network 側錄了這段請求，超過 5 分鐘後這段 JSON 就變成廢紙，GAS 會直接拒絕處理。

防止竄改資料 (Tampering)：駭客就算在 5 分鐘內攔截請求並把 UPDATE_ORDER 裡的金額改成 $1，因為他不知道 SECRET_KEY，無法重新計算出對應的 signature。只要字串被動過一個字，GAS 端算出來的 Hash 就對不起來，直接秒擋！

無 Base64 漏洞：完全捨棄 Base64，採用強加密標準的 Hex 輸出。
